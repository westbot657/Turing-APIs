#version 1.0.0

// :Global: is special case for top-level functions
// For other classes, :ClassName [opaque]: marks the class as opaque
// .  prefix denotes a variable, which currently are not fully implemented in every template.
// :: prefix denotes a static function for classes
// No prefix denotes an instance method except in top level where the prefix doesn't matter

:Log:
::Debug(msg: &str) -> void : _log__debug
::Critical(msg: &str) -> void : _log__critical
::Warn(msg: &str) -> void : _log__warn
::Info(msg: &str) -> void : _log__info

:Vivify:
::UploadBuffer(&Vu32) -> void : _vivify__upload_buffer
::GetBuffer() -> Vu32 : _vivify__get_buffer
